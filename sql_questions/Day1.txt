solution :



with q as (select
team_1 as team_win,
CASE
	WHEN team_1=winner THEN 1
	ELSE 0
END as win_count
from public.icc_world_cup
union ALL
select
team_2,
CASE
	WHEN team_2=winner THEN 1
	ELSE 0
END as win_count
from public.icc_world_cup
	)
select team_win,count(1),sum(win_count) as no_of_wins,count(1)-sum(win_count)
from q
group by team_win
order by no_of_wins
